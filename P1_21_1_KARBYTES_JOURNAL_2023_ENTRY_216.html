Write a C++ program which (a) instantiates an int array of 100 elements named A, (b) sets each element of A to some random nonnegative integer less than 100, (c) prints the array values to the command line, (d) uses bubble sort to arrange that array such that its elements are listed in ascending order (i.e. where A[0] is the smallest value and A[99] is the largest value), and (e) prints the sorted array values to the command line.

#include < iostream >
#include < cstdlib >
#include < ctime >

using namespace std;

int main()
{
    // Instantiate an int array of 100 elements named A
    int A[100];

    // Set each element of A to some random nonnegative integer less than 100
    srand(time(NULL));
    for (int i = 0; i < 100; i++)
    {
        A[i] = rand() % 100;
    }

    // Print the array values to the command line
    cout << "Unsorted array values:" << endl;
    for (int i = 0; i < 100; i++)
    {
        cout << A[i] << " ";
    }
    cout << endl;

    // Use bubble sort to arrange that array such that its elements are listed in ascending order
    int temp;
    for (int i = 0; i < 99; i++)
    {
        for (int j = 0; j < 99 - i; j++)
        {
            if (A[j] > A[j + 1])
            {
                temp = A[j];
                A[j] = A[j + 1];
                A[j + 1] = temp;
            }
        }
    }

    // Print the sorted array values to the command line
    cout << "Sorted array values:" << endl;
    for (int i = 0; i < 100; i++)
    {
        cout << A[i] << " ";
    }
    cout << endl;

    return 0;
}
